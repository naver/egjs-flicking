## Variable Size Panels
Flicking uses panel element's real size, and doesn't modify them.
<Flicking className="my-4" circular={true}>
  <Panel color="primary" style={{ width: "120px" }} index={0} />
  <Panel color="danger" style={{ width: "20%" }} index={1} />
  <Panel color="info" style={{ width: "500px" }} index={2} />
  <Panel color="warning" style={{ width: "300px" }} index={3} />
  <Panel color="success" style={{ width: "100%" }} index={4} />
</Flicking>
<SourceCode options={{ circular: true }} panels={[
  { tag: "div", style: { width: "120px" }, content: "1" },
  { tag: "div", style: { width: "20%" }, content: "2" },
  { tag: "div", style: { width: "500px" }, content: "3" },
  { tag: "div", style: { width: "300px" }, content: "4" },
  { tag: "div", style: { width: "100%" }, content: "5" }
]} />
